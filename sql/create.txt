CREATE DATABASE got

DEFAULT CHARACTER SET utf8

DEFAULT COLLATE utf8_unicode_ci;

CREATE TABLE characters (
id BIGINT NOT NULL,
dead BIT,
name VARCHAR(100) NOT NULL,
alias VARCHAR(100),
portrait LONGBLOB NOT NULL,
picture LOGBLOB NOT NULL,
bio VARCHAR(100),
organization VARCHAR(100),
house VARCHAR(100)
)
ENGINE = InnoDB;
